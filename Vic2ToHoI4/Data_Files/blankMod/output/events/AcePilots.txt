country_event = {
	id = ace_killed_by_ace.1
	title = ace_killed_by_ace.1.t
	desc = {
		text = ace_killed_by_ace.1.d
		#Don't show equipment name for countried with default names, e.g. Fighter I
		trigger = {
			OR = {
				NOT = {
					PREV = { 
						OWNER = {
						} 
					}
				}
				NOT = {
					FROM = { 
						OWNER = {
						}
					}
				}
			}
		}
	}
	desc = {
		text = ace_killed_by_ace.1.d_major
